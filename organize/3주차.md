# for문
```c
// for(초기값; 조건식; 증감) {
//     반복할 코드
// }
// 초기값 설정 -> 조건 -> 실행 -> 증감 -> 조건


// 예시코드
for(int i=1; i<=5; i++){
        for(int j=1; j<=i; j++){
            // j가 i와 같을 때까지로 조건을 주면
            // 1번째 줄은 한번, 2번째 줄은 두번 반복됨.
            printf("*");
        }
        printf("\n");
    }
/*
출력:
*
**
***
****
*****
*/
```
